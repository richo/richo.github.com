Allows reading from a rx.

Example

use std::io::ChanReader;

let (tx, rx) = channel();
let mut reader = ChanReader::new(rx);

let mut buf = ~[0u8, ..100];
match reader.read(buf) {
    Ok(nread) => println!("Read {} bytes", nread),
    Err(e) => println!("read error: {}", e),
}
